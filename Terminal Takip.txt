CREATE DATABASE TERMINALTAKIP;


CREATE TABLE TBLROUTE (
    ROUTEID INT IDENTITY (1,1),
    RLOCATION VARCHAR(100) NOT NULL,
	RNAME VARCHAR(100) NOT NULL,
    PRIMARY KEY (ROUTEID)
);

CREATE TABLE TBLDRIVER (
    DRIVERID INT IDENTITY (1,1),
    DNAME VARCHAR(50) NOT NULL,
    DSURNAME VARCHAR(50) NOT NULL,
    DGENDER VARCHAR(10) NOT NULL,
    DSALARY INT NOT NULL,
    PRIMARY KEY (DRIVERID)
);

CREATE TABLE TBLBUS (
    BUSID INT IDENTITY (1,1),
    BUSCAPACITY INT NOT NULL,
    DRIVER INT NOT NULL,
    TROUTE INT NOT NULL,
    DEPARTURETIME1 TIME(0) ,
    DEPARTURETIME2 TIME(0) ,
    DEPARTURETIME3 TIME(0) ,
    DEPARTURETIME4 TIME(0) ,
    PRIMARY KEY (BUSID),
    FOREIGN KEY (DRIVER) REFERENCES TBLDRIVER (DRIVERID),
    FOREIGN KEY (TROUTE) REFERENCES TBLROUTE (ROUTEID)
);



CREATE TABLE TBLSTATION (
    STATIONID INT IDENTITY (1,1),
    STATIONPOINT VARCHAR(100) NOT NULL,
    SROUTE INT NOT NULL,
    PRIMARY KEY (STATIONID),
    FOREIGN KEY (SROUTE) REFERENCES TBLROUTE (ROUTEID)
);


CREATE TABLE TBLPASSENGER (
    PASSENGERID INT IDENTITY (1,1),
    PNAME VARCHAR(50) NOT NULL,
    PSURNAME VARCHAR(50) NOT NULL,
    PGENDER VARCHAR(7) NOT NULL,
    PBIRTHDATE DATE NOT NULL,
    PFARE VARCHAR(50) NOT NULL,
    PRIMARY KEY (PASSENGERID)
);


CREATE TABLE TBLBOARDED (
    BUS INT NOT NULL,
    PASSENGER INT NOT NULL,
    PRIMARY KEY (BUS, PASSENGER),
    FOREIGN KEY (BUS) REFERENCES TBLBUS (BUSID),
    FOREIGN KEY (PASSENGER) REFERENCES TBLPASSENGER (PASSENGERID)
);

-- TBLPCOUNT tablosu olutur

CREATE TABLE TBLPCOUNT (
    BUS INT,  -- TBLBOARDED'daki BUS ile ilikilendirilmi
    PCOUNT INT
);


-----------------------------------------------------------------------------------------BOARDED tablosuna veri eklendiinde alacak trigger

CREATE TRIGGER UPDATE_PCOUNT ON TBLBOARDED
AFTER INSERT
AS
BEGIN
    -- TBLBOARDED tablosundaki BUS deerlerine gre grupla ve say
    DELETE FROM TBLPCOUNT;  -- ncelikle eski verileri temizle

    -- TBLPCOUNT tablosuna gruplanm ve saylm deerleri ekle
    INSERT INTO TBLPCOUNT (BUS, PCOUNT)
    SELECT BUS, COUNT(*) AS PCOUNT
    FROM TBLBOARDED
    GROUP BY BUS;
END;

-----------------------------------------------------------------------------------------------------------BOARDED tablosundan Veri silindiinde alacak trigger


CREATE TRIGGER DELETE_PCOUNT ON TBLBOARDED
AFTER DELETE
AS
BEGIN
    -- TBLBOARDED tablosundaki BUS deerlerine gre grupla ve say
    DELETE FROM TBLPCOUNT;  -- ncelikle eski verileri temizle

    -- TBLPCOUNT tablosuna gruplanm ve saylm deerleri ekle
    INSERT INTO TBLPCOUNT (BUS, PCOUNT)
    SELECT BUS, COUNT(*) AS PCOUNT
    FROM TBLBOARDED
    GROUP BY BUS;
END;

CREATE PROCEDURE GETBUSROUTE 
    @DEGER VARCHAR(100) = ' '
AS
    SELECT 
        BUSID, 
        BUSCAPACITY, 
        DEPARTURETIME1, 
        DEPARTURETIME2, 
        DEPARTURETIME3, 
        DEPARTURETIME4, 
        D.DNAME, 
        D.DSURNAME, 
        D.DGENDER, 
        R.RLOCATION, 
        R.RNAME 
    FROM TBLBUS B
    JOIN TBLROUTE R ON R.ROUTEID = B.TROUTE
    JOIN TBLDRIVER D ON D.DRIVERID = B.DRIVER
    WHERE R.RNAME = @DEGER;

-- Procedure' altrmak iin
EXEC GETBUSROUTE @DEGER = 'A2B';


------------------TBLSTATION Tablosunda Rota ismine gre Duraklar Getiren Procedure

CREATE PROCEDURE GETSTATION
    @DEGER VARCHAR(100) = ' '
AS
    SELECT 
        STATIONID, 
        STATIONPOINT, 
        R.RLOCATION, 
        R.RNAME 
    FROM TBLSTATION S
    JOIN TBLROUTE R ON R.ROUTEID = S.SROUTE
    WHERE R.RNAME = @DEGER;

-- Rota ismine gre o rotadaki duraklar getiren procedure
EXEC GETSTATION @DEGER = 'A8B';

-------------------TBLBOARDED ve TBLPASSENGER Tablosunda BUS a Gre Toplam Bilet cretini Hesaplayan Fonksiyon

CREATE FUNCTION get_total_fare (@bus INT)
RETURNS INT
AS
BEGIN
    DECLARE @total INT;
    SET @total = 0;
    -- TBLBOARDED ve TBLPASSENGER tablolarn birletirerek, verilen BUS deerine gre PFARE deerlerini al
    SELECT @total = @total + CASE PFARE
        WHEN 'STUDENT' THEN 8  -- STUDENT ise 8 ile arp
        WHEN 'STANDARD' THEN 15  -- STANDARD ise 15 ile arp
        ELSE 0  -- Dier durumlarda 0 ekle
    END
    FROM TBLBOARDED B
    INNER JOIN TBLPASSENGER P ON B.PASSENGER = P.PASSENGERID
    WHERE B.BUS = @bus;
    
    -- Toplam bilet cretini dndr
    RETURN @total;
END;
