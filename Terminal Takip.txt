CREATE DATABASE TERMINALTAKIP;


CREATE TABLE TBLROUTE (
    ROUTEID INT IDENTITY (1,1),
    RLOCATION VARCHAR(100) NOT NULL,
	RNAME VARCHAR(100) NOT NULL,
    PRIMARY KEY (ROUTEID)
);

CREATE TABLE TBLDRIVER (
    DRIVERID INT IDENTITY (1,1),
    DNAME VARCHAR(50) NOT NULL,
    DSURNAME VARCHAR(50) NOT NULL,
    DGENDER VARCHAR(10) NOT NULL,
    DSALARY INT NOT NULL,
    PRIMARY KEY (DRIVERID)
);

CREATE TABLE TBLBUS (
    BUSID INT IDENTITY (1,1),
    BUSCAPACITY INT NOT NULL,
    DRIVER INT NOT NULL,
    TROUTE INT NOT NULL,
    DEPARTURETIME1 TIME(0) ,
    DEPARTURETIME2 TIME(0) ,
    DEPARTURETIME3 TIME(0) ,
    DEPARTURETIME4 TIME(0) ,
    PRIMARY KEY (BUSID),
    FOREIGN KEY (DRIVER) REFERENCES TBLDRIVER (DRIVERID),
    FOREIGN KEY (TROUTE) REFERENCES TBLROUTE (ROUTEID)
);



CREATE TABLE TBLSTATION (
    STATIONID INT IDENTITY (1,1),
    STATIONPOINT VARCHAR(100) NOT NULL,
    SROUTE INT NOT NULL,
    PRIMARY KEY (STATIONID),
    FOREIGN KEY (SROUTE) REFERENCES TBLROUTE (ROUTEID)
);


CREATE TABLE TBLPASSENGER (
    PASSENGERID INT IDENTITY (1,1),
    PNAME VARCHAR(50) NOT NULL,
    PSURNAME VARCHAR(50) NOT NULL,
    PGENDER VARCHAR(7) NOT NULL,
    PBIRTHDATE DATE NOT NULL,
    PFARE VARCHAR(50) NOT NULL,
    PRIMARY KEY (PASSENGERID)
);


CREATE TABLE TBLBOARDED (
    BUS INT NOT NULL,
    PASSENGER INT NOT NULL,
    PRIMARY KEY (BUS, PASSENGER),
    FOREIGN KEY (BUS) REFERENCES TBLBUS (BUSID),
    FOREIGN KEY (PASSENGER) REFERENCES TBLPASSENGER (PASSENGERID)
);

-- TBLPCOUNT tablosu oluştur

CREATE TABLE TBLPCOUNT (
    BUS INT,  -- TBLBOARDED'daki BUS ile ilişkilendirilmiş
    PCOUNT INT
);


-----------------------------------------------------------------------------------------BOARDED tablosuna veri eklendiğinde çalışacak trigger

CREATE TRIGGER UPDATE_PCOUNT ON TBLBOARDED
AFTER INSERT
AS
BEGIN
    -- TBLBOARDED tablosundaki BUS değerlerine göre grupla ve say
    DELETE FROM TBLPCOUNT;  -- Öncelikle eski verileri temizle

    -- TBLPCOUNT tablosuna gruplanmış ve sayılmış değerleri ekle
    INSERT INTO TBLPCOUNT (BUS, PCOUNT)
    SELECT BUS, COUNT(*) AS PCOUNT
    FROM TBLBOARDED
    GROUP BY BUS;
END;

-----------------------------------------------------------------------------------------------------------BOARDED tablosundan Veri silindiğinde çalışacak trigger


CREATE TRIGGER DELETE_PCOUNT ON TBLBOARDED
AFTER DELETE
AS
BEGIN
    -- TBLBOARDED tablosundaki BUS değerlerine göre grupla ve say
    DELETE FROM TBLPCOUNT;  -- Öncelikle eski verileri temizle

    -- TBLPCOUNT tablosuna gruplanmış ve sayılmış değerleri ekle
    INSERT INTO TBLPCOUNT (BUS, PCOUNT)
    SELECT BUS, COUNT(*) AS PCOUNT
    FROM TBLBOARDED
    GROUP BY BUS;
END;

---------TBLBUS Tablosundan ROUTE değerine göre veri getiren bir Procedure Oluşturma
CREATE PROCEDURE GETBUSROUTE 
    @DEGER VARCHAR(100) = ' '
AS
    SELECT 
        BUSID, 
        BUSCAPACITY, 
        DEPARTURETIME1, 
        DEPARTURETIME2, 
        DEPARTURETIME3, 
        DEPARTURETIME4, 
        D.DNAME, 
        D.DSURNAME, 
        D.DGENDER, 
        R.RLOCATION, 
        R.RNAME 
    FROM TBLBUS B
    JOIN TBLROUTE R ON R.ROUTEID = B.TROUTE
    JOIN TBLDRIVER D ON D.DRIVERID = B.DRIVER
    WHERE R.RNAME = @DEGER;

-- Procedure'ü çalıştırmak için
EXEC GETBUSROUTE @DEGER = 'A2B';

CREATE PROCEDURE GETSTATION
    @DEGER VARCHAR(100) = ' '
AS
    SELECT 
        STATIONID, 
        STATIONPOINT, 
        R.RLOCATION, 
        R.RNAME 
    FROM TBLSTATION S
    JOIN TBLROUTE R ON R.ROUTEID = S.SROUTE
    WHERE R.RNAME = @DEGER;

-- Rota ismine göre o rotadaki durakları getiren procedure
EXEC GETSTATION @DEGER = 'A8B';

-------------------TBLBOARDED ve TBLPASSENGER Tablosunda BUS a Göre Toplam Bilet Ücretini Hesaplayan Fonksiyon

CREATE FUNCTION get_total_fare (@bus INT)
RETURNS INT
AS
BEGIN
    DECLARE @total INT;
    SET @total = 0;
    -- TBLBOARDED ve TBLPASSENGER tablolarını birleştirerek, verilen BUS değerine göre PFARE değerlerini al
    SELECT @total = @total + CASE PFARE
        WHEN 'STUDENT' THEN 8  -- STUDENT ise 8 ile çarp
        WHEN 'STANDARD' THEN 15  -- STANDARD ise 15 ile çarp
        ELSE 0  -- Diğer durumlarda 0 ekle
    END
    FROM TBLBOARDED B
    INNER JOIN TBLPASSENGER P ON B.PASSENGER = P.PASSENGERID
    WHERE B.BUS = @bus;
    
    -- Toplam bilet ücretini döndür
    RETURN @total;
END;

SELECT dbo.get_total_fare(12) AS TotalFare;  -- BUS ID sini girdiğinde o araca ait yolcuların toplam bilet fiyatını veren fonksiyon
